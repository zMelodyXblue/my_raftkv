# CMake 3.16 is what ships with Ubuntu 20.04 LTS.
# None of the features we use require a newer version.
cmake_minimum_required(VERSION 3.16)
project(my_raftkv LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)   # for clangd

# ── Build options ─────────────────────────────────────────────────
option(BUILD_TESTS    "Build unit tests"       ON)
option(BUILD_EXAMPLES "Build example programs" ON)

# ── Output directories ────────────────────────────────────────────
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)

# ── Dependencies ──────────────────────────────────────────────────
# Threads
find_package(Threads REQUIRED)

# Protobuf — there are TWO protobuf installations on this machine:
#   /usr/local  – manually compiled protoc 3.12.4 (no matching dev libs)
#   /usr        – apt protobuf-dev 3.6.1 (matches grpc++ 1.16.1)
# We MUST use the system (apt) version so that our generated code
# links against the same libprotobuf as grpc++.
# Pin the FindProtobuf hints before calling find_package.
set(Protobuf_INCLUDE_DIR  /usr/include              CACHE PATH     "" FORCE)
set(Protobuf_LIBRARY      /usr/lib/x86_64-linux-gnu/libprotobuf.so CACHE FILEPATH "" FORCE)
set(Protobuf_LITE_LIBRARY /usr/lib/x86_64-linux-gnu/libprotobuf-lite.so CACHE FILEPATH "" FORCE)
set(Protobuf_PROTOC_EXECUTABLE /usr/bin/protoc       CACHE FILEPATH "" FORCE)
find_package(Protobuf REQUIRED)
message(STATUS "Protobuf version : ${Protobuf_VERSION}")
message(STATUS "Protobuf includes: ${Protobuf_INCLUDE_DIRS}")
message(STATUS "protoc            : ${Protobuf_PROTOC_EXECUTABLE}")

# gRPC — Ubuntu 20.04 apt ships gRPC 1.16.1.
# We do NOT use pkg_check_modules to get include dirs, because the
# grpc++ .pc file adds /usr/local/include which shadows the system
# protobuf 3.6.1 headers with a manually-compiled 3.12.4 installation.
# Instead we set include dirs explicitly to /usr/include, locate the
# libraries via pkg-config (link flags only), and find the plugin.
find_package(PkgConfig REQUIRED)
pkg_check_modules(GRPC   REQUIRED grpc++)
pkg_check_modules(GRPC_CORE REQUIRED grpc)
find_program(GRPC_CPP_PLUGIN grpc_cpp_plugin
             HINTS /usr/bin /usr/local/bin
             REQUIRED)
# Ignore pkg-config include dirs entirely to avoid the path conflict.
set(GRPC_INCLUDE_DIRS "")
message(STATUS "gRPC version      : ${GRPC_VERSION}")
message(STATUS "grpc_cpp_plugin   : ${GRPC_CPP_PLUGIN}")

# Root cause fix: CMake strips /usr/include from -I flags because GCC
# lists it as a built-in system path. But GCC searches /usr/local/include
# BEFORE /usr/include in its built-in list, which makes the 3.12.4 headers
# shadow the 3.6.1 ones. Using add_compile_options adds an explicit -I flag
# that is processed BEFORE any built-in system path, overriding the order.
add_compile_options(-I/usr/include)

# spdlog — already installed via apt (libspdlog-dev)
find_package(spdlog REQUIRED)

# ── Proto code generation ─────────────────────────────────────────
set(PROTO_DIR     ${CMAKE_SOURCE_DIR}/proto)
set(PROTO_GEN_DIR ${CMAKE_BINARY_DIR}/generated)
file(MAKE_DIRECTORY ${PROTO_GEN_DIR})

# Helper: given a proto file, add a custom command that generates
# the protobuf + gRPC C++ stubs into PROTO_GEN_DIR.
# Sets <stem>_SRCS and <stem>_HDRS in the caller's scope.
function(generate_grpc_proto PROTO_FILE)
  get_filename_component(STEM ${PROTO_FILE} NAME_WE)

  set(PB_CC   ${PROTO_GEN_DIR}/${STEM}.pb.cc)
  set(PB_H    ${PROTO_GEN_DIR}/${STEM}.pb.h)
  set(GRPC_CC ${PROTO_GEN_DIR}/${STEM}.grpc.pb.cc)
  set(GRPC_H  ${PROTO_GEN_DIR}/${STEM}.grpc.pb.h)

  add_custom_command(
    OUTPUT  ${PB_CC} ${PB_H} ${GRPC_CC} ${GRPC_H}
    COMMAND ${Protobuf_PROTOC_EXECUTABLE}
            --proto_path=${PROTO_DIR}
            --cpp_out=${PROTO_GEN_DIR}
            --grpc_out=${PROTO_GEN_DIR}
            --plugin=protoc-gen-grpc=${GRPC_CPP_PLUGIN}
            ${PROTO_FILE}
    DEPENDS ${PROTO_FILE}
    COMMENT "Generating protobuf/gRPC C++ for ${STEM}.proto"
    VERBATIM
  )

  set(${STEM}_SRCS ${PB_CC} ${GRPC_CC} PARENT_SCOPE)
  set(${STEM}_HDRS ${PB_H}  ${GRPC_H}  PARENT_SCOPE)
endfunction()

generate_grpc_proto(${PROTO_DIR}/raft.proto)
generate_grpc_proto(${PROTO_DIR}/kv.proto)

# ── proto_gen library ─────────────────────────────────────────────
# A static library that contains all generated pb/grpc stubs.
# Every module that uses proto types links against this target.
add_library(proto_gen STATIC
  ${raft_SRCS}
  ${kv_SRCS}
)
target_include_directories(proto_gen PUBLIC
  ${PROTO_GEN_DIR}
  ${Protobuf_INCLUDE_DIRS}
  ${GRPC_INCLUDE_DIRS}
  /usr/include          # grpcpp/ headers live here for apt gRPC 1.16.1
)
target_link_libraries(proto_gen PUBLIC
  ${Protobuf_LIBRARIES}
  ${GRPC_LIBRARIES}
  ${GRPC_CORE_LIBRARIES}
  Threads::Threads
)
# Suppress warnings in generated code
target_compile_options(proto_gen PRIVATE -w)

# ── Source modules ────────────────────────────────────────────────
add_subdirectory(src)

# ── Tests ─────────────────────────────────────────────────────────
if(BUILD_TESTS)
  enable_testing()
  add_subdirectory(tests)
endif()
